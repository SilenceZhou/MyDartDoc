ByteData encodeMessage (dynamic message)
override

Encodes the specified message in binary.

Returns null if the message is null.

Implementation

@override
ByteData encodeMessage(dynamic message) {
  if (message == null)
    return null;
  final WriteBuffer buffer = WriteBuffer();
  writeValue(buffer, message);
  return buffer.done();
}
